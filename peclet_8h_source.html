1  /*
2  * @brief nsb-pcm peclet solves the Navier-Stokes-Boussinesq equations coupled with phase-change
3  *
4  * @detail
5  *
6  * Based on peclet, which was based on deal.II Tutorial 26 by Wolfgang Bangerth, Texas A&M University, 2013
7  *
8  * Some of the more notable extensions include:
9  * - Step time with Newton iterations; each Newton iteration requires a linear system solve
10  * - Assembles Newton linearized differential of the Navier-Stokes-Boussinesq equation with latent heat of phase-change
11  * - Supports time-dependent non-zero Dirichlet and Neumann boundary condition
12  * - Re-designed parmameter handling
13  * - Generalized boundary condition handling via the parameter input file
14  * - Writes FEFieldFunction to disk, and can read it from disk to initialize a restart
15  * - Extended the FEFieldFunction class for extrapolation
16  * - Added verification via Method of Manufactured Solutions (MMS) with error table based on approach from Tutorial 7
17  * - Added test suite using ctest and the standard deal.II approach
18  * - Added a parameteric sphere-cylinder grid
19  * - Added a boundary grid refinement routine
20  *
21  * @author Alexander Zimmerman <zimmerman@aices.rwth-aachen.de>, RWTH AAchen University, 2016
22  */

264  {

267

273  /*
274  Working with deal.II's Function class has been interesting, and I'm
275  sure many of my choices are unorthodox. The most important lesson learned has been that
276  a Function<dim>* can point to any class derived from Function<dim>. The general design pattern
277  then is to instantitate all of the functions that might be needed, and then to point to the ones
278  actually being used.
279  */
280  Functions::ParsedFunction<dim> parsed_initial_values_function(dim + 2);
281 
282  this->params = Parameters::read<dim>(
283  parameter_file,
284  this->source_function,
285  this->exact_solution_function,
286  parsed_initial_values_function,
287  this->boundary_functions);
288 
289  this->create_coarse_grid();
290 
291  /*
292  Generalizing the handling of auxiliary functions is complicated. In most cases one should be able to use a ParsedFunction, but the generality of Function<dim>* allows for a standard way to account for any possible derived class of Function<dim>.
293  For example this allows for....
294  - an optional initial values function that interpolates an old solution loaded from disk.
295  - flexibily implementing general boundary conditions
296  */
297 
298  // Initial values function
299 
300  Triangulation<dim> field_grid;
301  DoFHandler<dim> field_dof_handler(field_grid);
302  Vector<double> field_solution;
303 
304  if (this->params.initial_values.function_name != "interpolate_old_field")
305  { // This will write files that need to exist.
306  this->setup_system(true);
307  FEFieldTools::save_field_parts(this->triangulation, this->dof_handler, this->solution);
308  }
309 
310  FEFieldTools::load_field_parts(
311  field_grid,
312  field_dof_handler,
313  field_solution,
314  this->fe);
315 
316  MyFunctions::ExtrapolatedField<dim> field_function(
317  field_dof_handler,
318  field_solution);
319 
320 
321  if (this->params.initial_values.function_name == "interpolate_old_field")
322  {
323  this->initial_values_function_pointer = &field_function;
324  }
325  else if (this->params.initial_values.function_name == "parsed")
326  {
327  this->initial_values_function_pointer = &parsed_initial_values_function;
328  }
329 
330  // Attach manifolds
331 
332  assert(dim < 3); // @todo: 3D extension: For now the CylindricalManifold is being ommitted.
333  // deal.II makes is impractical for a CylindricalManifold to exist in 2D.
334  SphericalManifold<dim> spherical_manifold(this->spherical_manifold_center);
335 
336  for (unsigned int i = 0; i < manifold_ids.size(); i++)
337  {
338  if (manifold_descriptors[i] == "spherical")
339  {
340  this->triangulation.set_manifold(manifold_ids[i], spherical_manifold);
341  }
342  }
343 
344  // Run initial refinement cycles
345 
346  this->triangulation.refine_global(this->params.refinement.initial_global_cycles);
347 
348  Refinement::refine_mesh_near_boundaries(
349  this->triangulation,
350  this->params.refinement.boundaries_to_refine,
351  this->params.refinement.initial_boundary_cycles);
352 
353  // Initialize the linear system
354 
355  this->setup_system();
356 
357  Vector<double> tmp;
358  Vector<double> forcing_terms;
359 
360  // Iterate
361  unsigned int pre_refinement_step = 0;
362 
363 start_time_iteration:
364 
365  tmp.reinit(this->solution.size());
366 
367  VectorTools::interpolate(this->dof_handler,
368  *this->initial_values_function_pointer,
369  this->old_solution);
370 
371  this->solution = this->old_solution;
372  this->time_step_counter = 0;
373  this->time = 0;
374 
375  this->time_step_size = this->params.time.step_size;
376  if (this->time_step_size < EPSILON)
377  {
378  this->time_step_size = this->params.time.end_time/
379  pow(2., this->params.time.global_refinement_levels);
380  }
381  double Delta_t = this->time_step_size;
382 
383  this->write_solution();
384 
385  this->final_time_step = false;
386  this->output_this_step = true;
387 
388  do
389  {
390  ++this->time_step_counter;
391  this->time = Delta_t*time_step_counter; // Incrementing the time directly would accumulate errors
392 
393  // Set some flags that will control output for this step.
394  this->final_time_step = this->time > this->params.time.end_time - EPSILON;
395 
396  bool at_interval = false;
397 
398  if (this->params.output.time_step_interval == 1)
399  {
400  at_interval = true;
401  }
402  else if (this->params.output.time_step_interval != 0)
403  {
404  if ((this->time_step_counter % this->params.output.time_step_interval) == 0)
405  {
406  at_interval = true;
407  }
408  }
409  else
410  {
411  at_interval = false;
412  }
413 
414  if (at_interval)
415  {
416  this->output_this_step = true;
417  }
418  else
419  {
420  this->output_this_step = false;
421  }
422 
427  this->step_time();
428 
429  if ((this->time_step_counter == 1) &&
430  (pre_refinement_step < this->params.refinement.adaptive.initial_cycles))
431  {
432  this->adaptive_refine();
433  ++pre_refinement_step;
434 
435  tmp.reinit(this->solution.size());
436 
437  std::cout << std::endl;
438 
439  goto start_time_iteration;
440  }
441  else if ((this->time_step_counter > 0)
442  && (params.refinement.adaptive.interval > 0) // % 0 (mod 0) is undefined
443  && (this->time_step_counter % params.refinement.adaptive.interval == 0))
444  {
445  for (unsigned int cycle = 0;
446  cycle < params.refinement.adaptive.cycles_at_interval; cycle++)
447  {
448  this->adaptive_refine();
449  }
450  tmp.reinit(this->solution.size());
451 
452  }
453 
454  this->old_solution = this->solution;
455 
456  } while (!this->final_time_step);
457 
458  // Save data for FEFieldFunction so that it can be loaded for initialization
459  FEFieldTools::save_field_parts(triangulation, dof_handler, solution);
460 
461  // Write error table
462  if (this->params.verification.enabled)
463  {
464  this->write_verification_table();
465  }
466 
467  // Cleanup
468  this->triangulation.set_manifold(0);
469 
470  }
